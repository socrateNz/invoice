import { InvoiceData } from "@/types/invoice";
import { forwardRef } from "react";
import ClassicTemplate from "./templates/ClassicTemplate";
import ModernTemplate from "./templates/ModernTemplate";
import MinimalistTemplate from "./templates/MinimalistTemplate";
import CreativeTemplate from "./templates/CreativeTemplate";
import ElegantTemplate from "./templates/ElegantTemplate";
import StartupTemplate from "./templates/StartupTemplate";
import CorporateTemplate from "./templates/CorporateTemplate";
import EcoTemplate from "./templates/EcoTemplate";
import RetroTemplate from "./templates/RetroTemplate";
import FuturisticTemplate from "./templates/FuturisticTemplate";

interface InvoicePreviewProps {
  data: InvoiceData;
}

const MAX_WEIGHT_FIRST = 6;
const MAX_WEIGHT_NEXT = 10;

const getItemWeight = (item: any) => {
  let weight = 1;
  if (item.description) {
    const newlines = (item.description.match(/\n/g) || []).length;
    const chars = item.description.length;
    const calculatedLines = Math.max(newlines, Math.floor(chars / 60));
    weight += calculatedLines * 0.4;
  }
  return weight;
};

const InvoicePreview = forwardRef<HTMLDivElement, InvoicePreviewProps>(({ data }, ref) => {
  // Calculer le total global
  const grandTotal = data.items.reduce((acc, item) => acc + (parseFloat(item.quantity) || 1) * (item.price || 0), 0);

  // Chunking mathematical
  const pages: InvoiceData[] = [];
  let remaining = [...data.items];
  
  if (remaining.length === 0) {
    pages.push({ ...data, isFirstPage: true, isLastPage: true, pageIndex: 0, totalPages: 1, grandTotal });
  } else {
    let currentPage: any[] = [];
    let currentWeight = 0;
    let isFirst = true;

    remaining.forEach(item => {
      const weight = getItemWeight(item);
      const limit = isFirst ? MAX_WEIGHT_FIRST : MAX_WEIGHT_NEXT;

      if (currentWeight + weight > limit && currentPage.length > 0) {
        pages.push({ ...data, items: currentPage, grandTotal });
        currentPage = [item];
        currentWeight = weight;
        isFirst = false;
      } else {
        currentPage.push(item);
        currentWeight += weight;
      }
    });

    if (currentPage.length > 0) {
      pages.push({ ...data, items: currentPage, grandTotal });
    }
    
    // Assign pagination metadata
    pages.forEach((page, index) => {
      page.isFirstPage = index === 0;
      page.isLastPage = index === pages.length - 1;
      page.pageIndex = index;
      page.totalPages = pages.length;
    });
  }

  const renderTemplate = (pageData: InvoiceData) => {
    switch (data.template) {
      case 'modern': return <ModernTemplate data={pageData} />;
      case 'minimalist': return <MinimalistTemplate data={pageData} />;
      case 'creative': return <CreativeTemplate data={pageData} />;
      case 'elegant': return <ElegantTemplate data={pageData} />;
      case 'startup': return <StartupTemplate data={pageData} />;
      case 'corporate': return <CorporateTemplate data={pageData} />;
      case 'eco': return <EcoTemplate data={pageData} />;
      case 'retro': return <RetroTemplate data={pageData} />;
      case 'futuristic': return <FuturisticTemplate data={pageData} />;
      case 'classic':
      default: return <ClassicTemplate data={pageData} />;
    }
  };

  return (
    <div ref={ref} className="flex flex-col gap-8">
      {pages.map((pageData, i) => (
        <div key={i} className="invoice-page overflow-hidden bg-white shadow-lg mx-auto w-[210mm] h-[297mm] relative" style={{ padding: '0' }}>
          {renderTemplate(pageData)}
        </div>
      ))}
    </div>
  );
});

InvoicePreview.displayName = "InvoicePreview";
export default InvoicePreview;
