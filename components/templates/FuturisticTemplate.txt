import { InvoiceData } from "@/types/invoice";
import { formatCurrency, numberToWordsFR } from "@/utils/format";
import { forwardRef } from "react";

interface TemplateProps {
  data: InvoiceData;
}

const FuturisticTemplate = forwardRef<HTMLDivElement, TemplateProps>(({ data }, ref) => {
  const calculateTotal = () => data.items.reduce((acc, item) => acc + (parseFloat(item.quantity) || 1) * (item.price || 0), 0);
  const total = data.grandTotal ?? calculateTotal();
  const totalInWords = numberToWordsFR(total);

  return (
    <div className="bg-[#ffffff] shadow-lg mx-auto w-[210mm] min-h-[297mm] relative" style={{ padding: '0' }}>
      {/* Dark background for futuristic feel, but using light enough text for PDF printing if needed. Actually PDF works fine with dark mode if allowed. Let's make it a sleek dark theme. */}
      <div ref={ref} className="p-[40px] bg-[#020617] h-full w-full text-[#e2e8f0] text-sm border-l-4 border-[#06b6d4]" style={{ fontFamily: data.fontFamily || "Arial, sans-serif" }}>

        <div className="absolute top-10 right-10 opacity-20">
          <svg width="100" height="100" viewBox="0 0 100 100" fill="none" xmlns="http://www.w3.org/2000/svg">
            <circle cx="50" cy="50" r="49" stroke="#06b6d4" strokeWidth="2" strokeDasharray="4 4" />
            <circle cx="50" cy="50" r="30" stroke="#3b82f6" strokeWidth="1" />
          </svg>
        </div>

        <header className="flex justify-between items-start mb-16 relative z-10">
          <div className="w-[50%]">
            <h1 className="text-4xl font-black uppercase text-[#06b6d4] tracking-widest mb-1">{data.senderName || 'CYBER CORP'}</h1>
            <p className="text-xs font-mono text-[#3b82f6] mb-4 tracking-widest">&gt;_ {data.senderSlogan || 'System.out.println("Future")'}</p>
            <div className="text-xs space-y-1 text-[#94a3b8] font-mono">
              <p>EMAIL // {data.senderEmail || 'sys@cyber.com'}</p>
              <p>COMM  // {data.senderPhone || '+000 000 000 000'}</p>
            </div>
          </div>
          <div className="text-right w-[40%]">
            <h2 className="text-2xl font-bold text-[#e2e8f0] uppercase mb-4 tracking-[0.3em] border-b border-[#3b82f6] pb-2 inline-block">INVOICE_DATA</h2>
            <div className="text-xs font-mono text-[#94a3b8] space-y-2">
              <p className="flex justify-between"><span className="text-[#06b6d4]">ID:</span> <span className="text-[#e2e8f0]">{data.invoiceNumber || '---'}</span></p>
              <p className="flex justify-between"><span className="text-[#06b6d4]">TS:</span> <span className="text-[#e2e8f0]">{data.invoiceDate || '---'}</span></p>
              <p className="flex justify-between"><span className="text-[#06b6d4]">EXP:</span> <span className="text-[#e2e8f0]">{data.invoiceValidity || '---'}</span></p>
            </div>
          </div>
        </header>

        {data.isFirstPage && (
          <>
        <div className="flex gap-4 mb-10">
          <div className="w-[60%] bg-[#0f172a] p-5 border border-[#1e293b] rounded">
            <h3 className="text-[10px] font-mono text-[#06b6d4] uppercase tracking-widest mb-2">TARGET_ENTITY</h3>
            <h4 className="font-bold text-lg text-[#f8fafc]">{data.recipientName || 'Entity Name'}</h4>
            <p className="text-[#94a3b8] whitespace-pre-line mt-1 text-xs">{data.recipientAddress || 'Entity Address'}</p>
          </div>
          <div className="w-[40%] bg-[#0f172a] p-5 border border-[#1e293b] rounded">
            <h3 className="text-[10px] font-mono text-[#3b82f6] uppercase tracking-widest mb-2">OPERATION_SCOPE</h3>
            <p className="font-medium text-[#e2e8f0] text-sm leading-relaxed">{data.subject || 'Operation Subject'}</p>
          </div>
        </div>
        </>
        )}

        <table className="w-full mb-10 border-collapse">
          <thead>
            <tr>
              <th className="py-2 px-2 text-left text-[#06b6d4] font-mono text-[10px] uppercase tracking-widest border-b border-[#1e293b]">PARAM / DETAILS</th>
              <th className="py-2 px-2 text-center w-24 text-[#06b6d4] font-mono text-[10px] uppercase tracking-widest border-b border-[#1e293b]">VOL</th>
              <th className="py-2 px-2 text-right w-32 text-[#06b6d4] font-mono text-[10px] uppercase tracking-widest border-b border-[#1e293b]">CREDITS</th>
            </tr>
          </thead>
          <tbody>
            {data.items.length > 0 ? data.items.map((item) => (
              <tr key={item.id} className="hover:bg-[#0f172a]">
                <td className="py-4 px-2 border-b border-[#1e293b] align-top">
                  <div className="font-bold text-[#f8fafc] mb-1">{item.designation || 'Désignation'}</div>
                  <div className="text-[#64748b] text-xs whitespace-pre-line">{item.description}</div>
                </td>
                <td className="py-4 px-2 border-b border-[#1e293b] align-top text-center font-mono text-[#e2e8f0]">{item.quantity || '1'}</td>
                <td className="py-4 px-2 border-b border-[#1e293b] align-top text-right font-mono text-[#3b82f6]">{formatCurrency((parseFloat(item.quantity) || 1) * (item.price || 0))}</td>
              </tr>
            )) : (
              <tr><td colSpan={3} className="py-8 text-center text-[#475569] font-mono text-xs">NO_DATA_FOUND</td></tr>
            )}
          </tbody>
        </table>

        {data.isLastPage && (
          <>
        <div className="flex justify-end mb-12">
          <div className="w-[50%] bg-[#0f172a] border border-[#06b6d4] p-4 flex justify-between items-center relative overflow-hidden">
            <div className="absolute top-0 left-0 w-1 h-full bg-[#06b6d4]"></div>
            <span className="font-mono text-[#94a3b8] uppercase tracking-widest text-xs pl-2">TOTAL_REQ</span>
            <span className="font-black text-2xl text-[#f8fafc]">{formatCurrency(total)} <span className="text-xs text-[#06b6d4] font-mono">FCFA</span></span>
          </div>
        </div>

        <div className="mb-16 text-[#64748b] text-xs text-center font-mono uppercase tracking-widest border-y border-[#1e293b] py-3">
          CHECKSUM: <span className="text-[#e2e8f0]">{totalInWords} FCFA</span>
        </div>

        <div className="flex justify-between items-end mt-auto text-xs font-mono">
          <div>
            <p className="text-[#06b6d4] mb-1">TX_METHOD:</p>
            <p className="text-[#94a3b8]">BANK_TRANSFER // MOBILE_MONEY</p>
          </div>
          <div className="text-right">
            <p className="text-[#06b6d4] mb-6">AUTH_KEY:</p>
            <p className="text-[#e2e8f0]">{data.senderName ? data.senderName.toUpperCase() : 'SYSADMIN'}</p>
          </div>
        </div>
        </>)}

      </div>
    </div>
  );
});

FuturisticTemplate.displayName = "FuturisticTemplate";
export default FuturisticTemplate;
