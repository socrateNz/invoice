import { InvoiceData } from "@/types/invoice";
import { formatCurrency, numberToWordsFR } from "@/utils/format";
import { forwardRef } from "react";

interface TemplateProps {
  data: InvoiceData;
}

const MinimalistTemplate = forwardRef<HTMLDivElement, TemplateProps>(({ data }, ref) => {
  const calculateTotal = () => {
    return data.items.reduce((acc, item) => acc + (parseFloat(item.quantity) || 1) * (item.price || 0), 0);
  };

  const total = data.grandTotal ?? calculateTotal();
  const totalInWords = numberToWordsFR(total);

  return (
    <div className="bg-[#ffffff] shadow-lg mx-auto w-[210mm] min-h-[297mm] relative" style={{ padding: '0' }}>
      <div ref={ref} className="p-[40px] bg-[#ffffff] h-full w-full text-[#000000] text-sm" style={{ fontFamily: data.fontFamily || "Arial, Helvetica, sans-serif" }}>

        {/* Header */}
        <header className="flex justify-between items-start mb-16">
          <div className="w-[50%]">
            <h1 className="text-2xl font-bold uppercase tracking-widest mb-1">{data.senderName || 'NOM ENTREPRISE'}</h1>
            <p className="text-xs mb-1 opacity-80">{data.senderSlogan || 'Slogan de l\'entreprise'}</p>
            <p className="text-xs opacity-80">{data.senderEmail || 'email@exemple.com'} — {data.senderPhone || '+000 000 000 000'}</p>
          </div>
          <div className="text-right w-[40%]">
            <h2 className="text-2xl font-normal uppercase mb-2 tracking-widest border-b border-[#000000] pb-1 inline-block">Facture</h2>
            <div className="text-xs space-y-1 mt-2">
              <p>Facture N° : <span className="font-semibold">{data.invoiceNumber || '---'}</span></p>
              <p>Date : <span className="font-semibold">{data.invoiceDate || '---'}</span></p>
            </div>
          </div>
        </header>

        {/* Client Info */}
        {data.isFirstPage && (
          <>
        <div className="flex justify-end mb-16">
          <div className="w-[50%] border-l-2 border-[#000000] pl-4">
            <p className="text-xs uppercase tracking-widest mb-2 opacity-60">Facturé à</p>
            <h3 className="font-bold text-lg mb-1">{data.recipientName || 'Nom du Client'}</h3>
            <p className="whitespace-pre-line text-sm leading-relaxed">{data.recipientAddress || 'Adresse du client'}</p>
          </div>
        </div>

        {/* Subject */}
        <div className="mb-8 pb-2 border-b border-[#000000]">
          <span className="font-bold text-xs uppercase tracking-widest mr-2">Objet :</span>
          <span>{data.subject || 'Objet de la facture'}</span>
        </div>
        </>)}

        {/* Items Table */}
        <table className="w-full mb-12">
          <thead>
            <tr>
              <th className="py-2 text-left border-b-2 border-[#000000] font-bold text-xs uppercase tracking-wider">Description</th>
              <th className="py-2 text-center w-24 border-b-2 border-[#000000] font-bold text-xs uppercase tracking-wider">Qté</th>
              <th className="py-2 text-right w-32 border-b-2 border-[#000000] font-bold text-xs uppercase tracking-wider">Montant</th>
            </tr>
          </thead>
          <tbody>
            {data.items.length > 0 ? (
              data.items.map((item) => (
                <tr key={item.id}>
                  <td className="py-4 align-top border-b border-[#e5e7eb]">
                    <div className="font-bold mb-1">{item.designation || 'Désignation'}</div>
                    <div className="text-xs whitespace-pre-line opacity-80">{item.description}</div>
                  </td>
                  <td className="py-4 align-top text-center border-b border-[#e5e7eb]">
                    {item.quantity || '1'}
                  </td>
                  <td className="py-4 align-top text-right border-b border-[#e5e7eb] font-medium">
                    {formatCurrency((parseFloat(item.quantity) || 1) * (item.price || 0))}
                  </td>
                </tr>
              ))
            ) : (
              <tr>
                <td colSpan={3} className="py-8 text-center text-[#9ca3af] border-b border-[#e5e7eb]">
                  Aucune ligne ajoutée
                </td>
              </tr>
            )}
          </tbody>
        </table>

        {/* Total Box */}
        {data.isLastPage && (
          <>
        <div className="flex justify-end mb-8">
          <div className="w-[40%]">
            <div className="flex justify-between items-center py-2 border-t-2 border-b-2 border-[#000000]">
              <span className="font-bold uppercase tracking-widest text-xs">Total</span>
              <span className="font-bold text-xl">{formatCurrency(total)} FCFA</span>
            </div>
          </div>
        </div>

        {/* Amount in words */}
        <div className="mb-16 text-sm text-center">
          <p>Arrêtée à la somme de <span className="font-bold uppercase">{totalInWords} Francs CFA</span>.</p>
        </div>

        {/* Footer info */}
        <div className="mt-auto grid grid-cols-2 gap-8 text-xs">
          <div>
            <p className="font-bold uppercase tracking-widest mb-1">Règlement</p>
            <p className="opacity-80">Virement bancaire ou Mobile Money</p>
          </div>
          <div className="text-right">
            <p className="font-bold uppercase tracking-widest mb-1">Signature</p>
            <p className="opacity-80">{data.senderName ? data.senderName.toUpperCase() : 'L\'ENTREPRISE'}</p>
          </div>
        </div>
        </>)}

      </div>
    </div>
  );
});

MinimalistTemplate.displayName = "MinimalistTemplate";
export default MinimalistTemplate;
