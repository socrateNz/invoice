import { InvoiceData } from "@/types/invoice";
import { formatCurrency, numberToWordsFR } from "@/utils/format";
import { forwardRef } from "react";

interface TemplateProps {
  data: InvoiceData;
}

const ModernTemplate = forwardRef<HTMLDivElement, TemplateProps>(({ data }, ref) => {
  const calculateTotal = () => {
    return data.items.reduce((acc, item) => acc + (parseFloat(item.quantity) || 1) * (item.price || 0), 0);
  };

  const total = data.grandTotal ?? calculateTotal();
  const totalInWords = numberToWordsFR(total);

  return (
    <div className="bg-[#ffffff] shadow-lg mx-auto w-[210mm] min-h-[297mm] relative" style={{ padding: '0' }}>
      <div ref={ref} className="p-[40px] bg-[#ffffff] h-full w-full text-[#374151] text-sm" style={{ fontFamily: data.fontFamily || "Arial, Helvetica, sans-serif" }}>

        {/* Header */}
        <header className="flex justify-between items-start mb-12 border-b-2 border-[#1f2937] pb-8">
          <div className="w-[50%]">
            <h1 className="text-4xl font-bold uppercase tracking-tight text-[#111827] mb-2">{data.senderName || 'NOM ENTREPRISE'}</h1>
            <p className="text-sm mb-4 text-[#4b5563]">{data.senderSlogan || 'Slogan de l\'entreprise'}</p>
            <div className="text-xs space-y-1 text-[#6b7280]">
              <p>{data.senderEmail || 'email@exemple.com'}</p>
              <p>{data.senderPhone || '+000 000 000 000'}</p>
            </div>
          </div>
          <div className="text-right w-[40%]">
            <h2 className="text-3xl font-light text-[#9ca3af] uppercase mb-4 tracking-widest">Facture</h2>
            <div className="text-sm text-[#4b5563] space-y-2">
              <div className="flex justify-end gap-4"><span className="font-semibold text-[#111827]">N°</span> <span>{data.invoiceNumber || '---'}</span></div>
              <div className="flex justify-end gap-4"><span className="font-semibold text-[#111827]">Date</span> <span>{data.invoiceDate || '---'}</span></div>
              <div className="flex justify-end gap-4"><span className="font-semibold text-[#111827]">Validité</span> <span>{data.invoiceValidity || '---'}</span></div>
            </div>
          </div>
        </header>

        {/* Client Info */}
        {data.isFirstPage && (
          <>
        <div className="flex mb-10">
          <div className="w-1/2">
            <h3 className="text-xs font-bold text-[#9ca3af] uppercase tracking-wider mb-2">Facturé à</h3>
            <h4 className="font-bold text-lg text-[#111827]">{data.recipientName || 'Nom du Client'}</h4>
            <p className="text-[#4b5563] whitespace-pre-line mt-1">{data.recipientAddress || 'Adresse du client'}</p>
          </div>
          <div className="w-1/2 pl-8 border-l border-[#e5e7eb]">
            <h3 className="text-xs font-bold text-[#9ca3af] uppercase tracking-wider mb-2">Objet</h3>
            <p className="font-medium text-[#1f2937]">{data.subject || 'Objet de la facture'}</p>
          </div>
        </div>
        </>)}

        {/* Items Table */}
        <table className="w-full mb-8">
          <thead>
            <tr className="border-b-2 border-[#111827]">
              <th className="py-3 px-2 text-left w-12 text-[#111827] font-bold text-xs uppercase">N°</th>
              <th className="py-3 px-2 text-left text-[#111827] font-bold text-xs uppercase">Désignation</th>
              <th className="py-3 px-2 text-center w-24 text-[#111827] font-bold text-xs uppercase">Qté</th>
              <th className="py-3 px-2 text-right w-32 text-[#111827] font-bold text-xs uppercase">Montant (FCFA)</th>
            </tr>
          </thead>
          <tbody>
            {data.items.length > 0 ? (
              data.items.map((item, index) => (
                <tr key={item.id} className="border-b border-[#e5e7eb]">
                  <td className="py-4 px-2 align-top text-center text-[#6b7280]">
                    {(index + 1).toString().padStart(2, '0')}
                  </td>
                  <td className="py-4 px-2 align-top">
                    <div className="font-bold text-[#111827] mb-1">{item.designation || 'Désignation'}</div>
                    <div className="text-[#6b7280] text-xs whitespace-pre-line">{item.description}</div>
                  </td>
                  <td className="py-4 px-2 align-top text-center text-[#4b5563]">
                    {item.quantity || '1'}
                  </td>
                  <td className="py-4 px-2 align-top text-right font-medium text-[#111827]">
                    {formatCurrency((parseFloat(item.quantity) || 1) * (item.price || 0))}
                  </td>
                </tr>
              ))
            ) : (
              <tr>
                <td colSpan={4} className="py-8 text-center text-[#9ca3af] border-b border-[#e5e7eb]">
                  Aucune ligne ajoutée
                </td>
              </tr>
            )}
          </tbody>
        </table>

        {/* Total Box */}
        {data.isLastPage && (
          <>
        <div className="flex justify-end mb-10">
          <div className="w-1/2">
            <div className="flex justify-between items-center bg-[#f3f4f6] p-4 rounded-lg">
              <span className="font-bold text-[#111827] uppercase tracking-wider text-sm">Total Général</span>
              <span className="font-bold text-2xl text-[#111827]">{formatCurrency(total)} FCFA</span>
            </div>
          </div>
        </div>

        {/* Amount in words */}
        <div className="mb-12 text-[#6b7280] italic text-sm">
          Arrêtée la présente facture à la somme de : <span className="font-semibold text-[#111827]">{totalInWords} Francs CFA</span>
        </div>

        {/* Footer info */}
        <div className="flex justify-between mt-auto">
          <div className="w-1/2">
            <h4 className="font-bold text-[#111827] text-xs uppercase mb-2">Conditions</h4>
            <p className="text-xs text-[#6b7280]">Mode de règlement : Virement bancaire ou Mobile Money</p>
          </div>
          <div className="w-1/2 text-right relative h-32">
            <h4 className="font-bold text-[#111827] text-xs uppercase mb-1">La Direction</h4>
            <p className="text-xs text-[#6b7280] mb-8">{data.senderName ? data.senderName.toUpperCase() : 'L\'ENTREPRISE'}</p>
          </div>
        </div>
        </>)}

        {/* Absolute Footer line */}
        <div className="absolute bottom-10 left-[40px] right-[40px]">
          <div className="border-t border-[#e5e7eb] pt-4 text-center text-xs text-[#9ca3af] flex justify-between">
            <span className="font-bold text-[#4b5563]">{data.senderName ? data.senderName.toUpperCase() : 'ETARCOS DEV'}</span>
            <span>{data.senderEmail || 'email@exemple.com'} | {data.senderPhone || '+000 000 000 000'}</span>
          </div>
        </div>

      </div>
    </div>
  );
});

ModernTemplate.displayName = "ModernTemplate";
export default ModernTemplate;
