import { InvoiceData } from "@/types/invoice";
import { formatCurrency, numberToWordsFR } from "@/utils/format";
import { forwardRef } from "react";

interface TemplateProps {
  data: InvoiceData;
}

const CorporateTemplate = forwardRef<HTMLDivElement, TemplateProps>(({ data }, ref) => {
  const calculateTotal = () => data.items.reduce((acc, item) => acc + (parseFloat(item.quantity) || 1) * (item.price || 0), 0);
  const total = data.grandTotal ?? calculateTotal();

  return (
    <div className="bg-[#ffffff] shadow-lg mx-auto w-[210mm] min-h-[297mm] relative" style={{ padding: '0' }}>
      <div ref={ref} className="p-[40px] bg-[#ffffff] h-full w-full text-[#334155] text-sm" style={{ fontFamily: data.fontFamily || "Arial, sans-serif" }}>

        {/* Corporate Header */}
        <header className="border-b-4 border-[#0f172a] pb-8 mb-10 flex justify-between items-end">
          <div className="w-[60%]">
            <h1 className="text-3xl font-black uppercase text-[#0f172a] tracking-tight">{data.senderName || 'ENTERPRISE LLC'}</h1>
            <p className="text-sm font-semibold text-[#64748b] mt-1">{data.senderSlogan || 'Global Solutions'}</p>
          </div>
          <div className="text-right w-[40%]">
            <h2 className="text-2xl font-bold text-[#0f172a] uppercase mb-1">Facture Commerciale</h2>
            <p className="text-[#64748b] text-sm">{data.invoiceNumber || '---'}</p>
          </div>
        </header>

        {/* Addresses Box */}
        {data.isFirstPage && (
          <>
        <div className="flex bg-[#f8fafc] border border-[#e2e8f0] mb-10">
          <div className="w-1/2 p-6 border-r border-[#e2e8f0]">
            <h3 className="text-xs font-bold text-[#0f172a] uppercase tracking-widest border-b border-[#cbd5e1] pb-2 mb-3">Émetteur</h3>
            <p className="font-bold text-[#0f172a]">{data.senderName}</p>
            <p className="text-[#475569] mt-2">{data.senderEmail || 'email@exemple.com'}</p>
            <p className="text-[#475569]">{data.senderPhone || '+000 000 000 000'}</p>
          </div>
          <div className="w-1/2 p-6">
            <h3 className="text-xs font-bold text-[#0f172a] uppercase tracking-widest border-b border-[#cbd5e1] pb-2 mb-3">Destinataire</h3>
            <p className="font-bold text-[#0f172a]">{data.recipientName || 'Client Name'}</p>
            <p className="text-[#475569] mt-2 whitespace-pre-line">{data.recipientAddress || 'Client Address'}</p>
          </div>
        </div>

        {/* Invoice Details */}
        <div className="flex mb-10 border-t border-b border-[#e2e8f0] py-4">
          <div className="w-1/3">
            <span className="text-xs text-[#64748b] uppercase font-bold block mb-1">Date d'émission</span>
            <span className="font-semibold text-[#0f172a]">{data.invoiceDate || '---'}</span>
          </div>
          <div className="w-1/3 border-l border-[#e2e8f0] pl-6">
            <span className="text-xs text-[#64748b] uppercase font-bold block mb-1">Échéance</span>
            <span className="font-semibold text-[#0f172a]">{data.invoiceValidity || '---'}</span>
          </div>
          <div className="w-1/3 border-l border-[#e2e8f0] pl-6">
            <span className="text-xs text-[#64748b] uppercase font-bold block mb-1">Objet</span>
            <span className="font-semibold text-[#0f172a]">{data.subject || '---'}</span>
          </div>
        </div>
        </>
        )}

        {/* Items Table */}
        <table className="w-full mb-10 border border-[#e2e8f0]">
          <thead>
            <tr className="bg-[#0f172a] text-[#ffffff]">
              <th className="py-3 px-4 text-left text-xs font-bold uppercase tracking-wider border-r border-[#334155]">Description</th>
              <th className="py-3 px-4 text-center w-24 text-xs font-bold uppercase tracking-wider border-r border-[#334155]">Qté</th>
              <th className="py-3 px-4 text-right w-32 text-xs font-bold uppercase tracking-wider">Montant</th>
            </tr>
          </thead>
          <tbody>
            {data.items.length > 0 ? data.items.map((item, i) => (
              <tr key={item.id} className={i % 2 === 0 ? "bg-[#ffffff]" : "bg-[#f8fafc]"}>
                <td className="py-4 px-4 border-b border-r border-[#e2e8f0] align-top">
                  <div className="font-bold text-[#0f172a] mb-1">{item.designation || 'Désignation'}</div>
                  <div className="text-[#64748b] text-xs whitespace-pre-line">{item.description}</div>
                </td>
                <td className="py-4 px-4 border-b border-r border-[#e2e8f0] align-top text-center font-medium">{item.quantity || '1'}</td>
                <td className="py-4 px-4 border-b border-[#e2e8f0] align-top text-right font-bold text-[#0f172a]">{formatCurrency((parseFloat(item.quantity) || 1) * (item.price || 0))}</td>
              </tr>
            )) : (
              <tr><td colSpan={3} className="py-8 text-center text-[#cbd5e1] border-b border-[#e2e8f0]">Aucune ligne</td></tr>
            )}
          </tbody>
        </table>

        {/* Total Box */}
        {data.isLastPage && (
          <>
        <div className="flex justify-end mb-10">
          <div className="w-[40%] bg-[#f8fafc] border border-[#0f172a]">
            <div className="bg-[#0f172a] text-[#ffffff] p-3 text-center">
              <span className="font-bold uppercase tracking-wider text-sm">Total à payer</span>
            </div>
            <div className="p-4 text-center">
              <span className="font-black text-2xl text-[#0f172a]">{formatCurrency(total)} FCFA</span>
            </div>
          </div>
        </div>

        <div className="mb-16 text-[#64748b] text-sm text-center">
          La somme due est de <span className="font-bold text-[#0f172a] uppercase">{numberToWordsFR(total)} Francs CFA</span>.
        </div>

        <div className="flex justify-between items-end mt-auto text-xs border-t-2 border-[#e2e8f0] pt-6">
          <div>
            <p className="font-bold text-[#0f172a] mb-1 uppercase tracking-widest">Modalités de paiement</p>
            <p className="text-[#64748b]">Virement bancaire / Mobile Money</p>
          </div>
          <div className="text-right">
            <p className="font-bold text-[#0f172a] mb-8 uppercase tracking-widest">Visa / Signature</p>
            <p className="text-[#64748b] font-medium">{data.senderName ? data.senderName.toUpperCase() : 'L\'ENTREPRISE'}</p>
          </div>
        </div>
        </>)}

      </div>
    </div>
  );
});

CorporateTemplate.displayName = "CorporateTemplate";
export default CorporateTemplate;
